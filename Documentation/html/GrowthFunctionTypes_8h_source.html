292  GridBasedGrowthFunction(int id, int type, float initTime, float endTime, bool applyToColumnarLayer, bool applyToPeripodialMembrane, bool applyToBasalECM, bool applyToLateralECM, int nX, int nY, std::vector<std::vector<std::array<double,3>>>& GrowthMat, std::vector<std::vector<double>>& AngleMat) : GrowthFunctionBase(id, type, initTime, endTime, applyToColumnarLayer, applyToPeripodialMembrane, applyToBasalECM, applyToLateralECM){
299  double aspectRatioThreshold = 1.1;
300  this ->nGridX = nX;
301  this ->nGridY = nY;
302  //GrowthMatrix : [grid_i] [grid_j][x,y,z]
303  for (size_t i= 0; i<nGridX; ++i){
304  std::vector<std::array<double,3>> tmpGridMatrixY(nGridY,std::array<double,3>{0.0});
305  GrowthMatrix.push_back(tmpGridMatrixY);
306  std::vector<double> tmpShearY(nGridY,0.0);
307  xyShearAngleMatrix.push_back(tmpShearY);
308  std::vector<bool> tmpAROverThresY (nGridY,true); //set AR to be significant, and assign it to be too small if necesarry
309  aspectRatioOverThresoldMatrix.push_back(tmpAROverThresY);
310 
311  //matrices with 4 corners,for compatable averaging:
312  std::vector<std::array<std::array<double,3>,4>> tmpGrid4CornerMatrixY(nGridY,std::array<std::array<double,3>,4>{std::array<double,3>{0.0}});
313  compatibleGrowths.push_back(tmpGrid4CornerMatrixY);
314  std::vector<std::array<bool,4>> tmpGrid4CornerAngleEliminatedY(nGridY,std::array<bool,4>{false});
315  compatibleAngleEliminated.push_back(tmpGrid4CornerAngleEliminatedY);
316  std::vector<std::array<double,4>> tmpGrid4CornerAngleY(nGridY,std::array<double,4>{0.0});
317  compatibleAngles.push_back(tmpGrid4CornerAngleY);
318  }
319  for (size_t i=0; i<nGridX; ++i){
320  for (size_t j=0; j<nGridY; ++j){
321  xyShearAngleMatrix[i][j] = AngleMat[i][j]*M_PI/180.0; //converting to radians
322  for (size_t k=0; k<3; ++k){
323  GrowthMatrix[i][j][k] = GrowthMat[i][j][k];
324  }
325  double AR = GrowthMatrix[i][j][0] / GrowthMatrix[i][j][1];
326  if (AR < 0 ){
327  //some of the growth rates have higher aspect ratio than their nuclei count, for instance, an aspect ratio of 5 with nuclei count 4
328  //then the tissue actually needs to actively shring in one direction while growing in the other.
329  //this will lead to a negative aspect ratio, and this extremely high aspect ratio would be eliminated by the threshold, as it would be less than the positive threshold.
330  AR *=-1.0;
331  }
332  if (AR < 1.0){
333  AR = 1/AR;
334  }
335  if(AR < aspectRatioThreshold){
336  aspectRatioOverThresoldMatrix[i][j] = false;
337  }
338  }
339  }
340 
341  xyShearRotationsMatrix = new gsl_matrix**[(const int) nGridX];
342  for (size_t i=0; i<nGridX; ++i){
343  xyShearRotationsMatrix[i] = new gsl_matrix*[(const int) nGridY];
344  for (size_t j=0; j<nGridY; ++j){
345  xyShearRotationsMatrix[i][j] = new gsl_matrix;
346  xyShearRotationsMatrix[i][j] = gsl_matrix_calloc(3,3);
347  double c = cos(xyShearAngleMatrix[i][j]);
348  double s = sin(xyShearAngleMatrix[i][j]);
349  //cout<<"xyShearRotationsMatrix ["<<i<<"]["<<j<<"], angle: "<<xyShearAngleMatrix[i][j]<<" cos: "<<c<<" sin "<<s<<std::endl;
350  gsl_matrix_set(xyShearRotationsMatrix[i][j],0,0, c );
351  gsl_matrix_set(xyShearRotationsMatrix[i][j],0,1, -1.0*s);
352  gsl_matrix_set(xyShearRotationsMatrix[i][j],0,2, 0.0);
353  gsl_matrix_set(xyShearRotationsMatrix[i][j],1,0, s);
354  gsl_matrix_set(xyShearRotationsMatrix[i][j],1,1, c);
355  gsl_matrix_set(xyShearRotationsMatrix[i][j],1,2, 0.0);
356  gsl_matrix_set(xyShearRotationsMatrix[i][j],2,0, 0.0);
357  gsl_matrix_set(xyShearRotationsMatrix[i][j],2,1, 0.0);
358  gsl_matrix_set(xyShearRotationsMatrix[i][j],2,2, 1.0);
359  }

362  preCalculateAnglesForCompatibleAveraging();

403  void preCalculateAnglesForCompatibleAveraging(){
404  for (size_t IndexX = 0; IndexX<nGridX-1; ++IndexX){
405  for (size_t IndexY = 0; IndexY<nGridY-1; ++IndexY){
406  //read the growth rates at four corners for this grid point:
407  double growth0[3] = {0.0,0.0,0.0};
408  double growth1[3] = {0.0,0.0,0.0};
409  double growth2[3] = {0.0,0.0,0.0};
410  double growth3[3] = {0.0,0.0,0.0};
411  for (int axis =0; axis<3; axis++){
412  growth0[axis] = getGrowthMatrixElement(IndexX,IndexY,axis);
413  growth1[axis] = getGrowthMatrixElement(IndexX+1,IndexY,axis);
414  growth2[axis] = getGrowthMatrixElement(IndexX,IndexY+1,axis);
415  growth3[axis] = getGrowthMatrixElement(IndexX+1,IndexY+1,axis);
416  }
417  //now reading the angles:
418  double angles[4] = {0.0,0.0,0.0,0.0};
419  //Not summing aspect ratios close to one:
420  //and recording the eliminated angles for ease of use later on:
421  bool AngleEliminated[4] = {false, false, false, false};
422  if (isAspectRatioOverOne( IndexX, IndexY)) {angles[0] = getXyShearAngleMatrixElement(IndexX , IndexY); }
423  else {AngleEliminated[0] = true;}
424  if (isAspectRatioOverOne( IndexX+1, IndexY)) {angles[1] = getXyShearAngleMatrixElement(IndexX+1, IndexY); }
425  else {AngleEliminated[1] = true;}
426  if (isAspectRatioOverOne( IndexX, IndexY+1)) {angles[2] = getXyShearAngleMatrixElement(IndexX, IndexY+1);}
427  else {AngleEliminated[2] = true;}
428  if (isAspectRatioOverOne( IndexX+1, IndexY+1)) {angles[3] = getXyShearAngleMatrixElement(IndexX+1, IndexY+1);}
429  else {AngleEliminated[3] = true;}
430  //bring all angles in the interval 0 - 180:
431  for (int i=0; i<4; ++i){
432  while (angles[i]<0){angles[i] += M_PI;}
433  while (angles[i]>M_PI){angles[i] -= M_PI;}
434  }
435  //now I am bringing the angles to the vicinity of each other:
436  //selecting the base angle, the first one that was not eliminated will do:
437  bool angleSelected = false;
438  int selectedIndice = 3;
439  double tet = 0;
440  for (int i=0; i<4; ++i){
441  if(!AngleEliminated[i] && !angleSelected){
442  tet = angles[i];
443  angleSelected = true;
444  }
445  if (angleSelected){
446  selectedIndice = i;
447  break;
448  }
449  }
450  //selected the first angle, now bring all the rest near this one:
451  //I keep a record if I need to flip the growths:
452  bool flipGrowth[4] = {false, false, false, false};
453  for (int i=selectedIndice; i<4; ++i){
454  double dtet = angles[i] - tet; // angle from selected angle (tet) to current angle (angles[i]).
455  if (dtet>0){
456  //selected tetha is larger than this angle:
457  if (dtet >= M_PI/4.0 && dtet < 3.0*M_PI/4.0 ){
458  //the difference is between 45-135 degrees, I flip the growth, and rotate angle -90 degrees.
459  //The difference comes to the interval (-45) - 45
460  angles[i] = angles[i] - M_PI/2.0;
461  flipGrowth[i] = true;
462  }
463  else if (dtet >= 3.0*M_PI/4.0 && dtet< 5.0*M_PI/4.0 ){
464  //the difference is between 135-225 degrees, I only need to rotate -180 degrees,
465  //no flip necessary, the angle will fall into the interval (-45) - 45
466  angles[i] = angles[i] - M_PI;
467  }
468  else if (dtet >= 5.0*M_PI/4.0 && dtet< 7.0*M_PI/4.0){
469  //the difference is between 225 and 315 degrees, I need to rotate and flip as in the first case.
470  //I need to rotate by -270 degrees to bring to the interval (-45) - 45
471  angles[i] = angles[i] - 6.0*M_PI/4.0;
472  flipGrowth[i] = true;
473  }
474  else if (dtet >= 7.0*M_PI/4.0){
475  //the difference is higher than 315 degrees. I need to rotate the angle by -360 degrees, and
476  //the difference will come into (-45) - 0 (as it cannot be higher than 360. No need to flip
477  angles[i] = angles[i] - 2.0*M_PI;
478  }
479  }
480  else if (dtet<0){
481  //selected tetha is smaller than this angle:
482  if (dtet <= -M_PI/4.0 && dtet > -3.0*M_PI/4.0 ){
483  //the difference is between (-45)- (-135) degrees, I flip the growth, and rotate angle +90 degrees.
484  //The difference comes to the interval (-45) - 45
485  angles[i] = angles[i] + M_PI/2.0;
486  flipGrowth[i] = true;
487  }
488  else if (dtet <= -3.0*M_PI/4.0 && dtet > -5.0*M_PI/4.0 ){
489  //the difference is between (-135) - (-225) degrees, I only need to rotate +180 degrees,
490  //no flip necessary, the angle will fall into the interval (-45) - 45
491  angles[i] = angles[i] + M_PI;
492  }
493  else if (dtet <= -5.0*M_PI/4.0 && dtet > -7.0*M_PI/4.0){
494  //the difference is between (-225) and (-315) degrees, I need to rotate and flip as in the first case.
495  //I need to rotate by +270 degrees to bring to the interval (-45) - 45
496  angles[i] = angles[i] + 6.0*M_PI/4.0;
497  flipGrowth[i] = true;
498  }
499  else if (dtet <= -7.0*M_PI/4.0){
500  //the difference is less than (-315) degrees. I need to rotate the angle by +360 degrees, and
501  //the difference will come into 0 - 45 (as it cannot be lower than (-360). No need to flip
502  angles[i] = angles[i] + 2.0*M_PI;
503  }
504  }
505  }
506  //all angles are brought to the vicinity of the selected base angle;
507  //flipping the growth where necessary:
508  if (flipGrowth[0]){
509  double x = growth0[0];
510  growth0[0] = growth0[1];
511  growth0[1] = x;
512  }
513  if (flipGrowth[1]){
514  double x = growth1[0];
515  growth1[0] = growth1[1];
516  growth1[1] = x;
517  }
518  if (flipGrowth[2]){
519  double x = growth2[0];
520  growth2[0] = growth2[1];
521  growth2[1] = x;
522  }
523  if (flipGrowth[3]){
524  double x = growth3[0];
525  growth3[0] = growth3[1];
526  growth3[1] = x;
527  }
528  for (size_t axis=0; axis<3; ++axis){
529  compatibleGrowths[IndexX][IndexY][0][axis]= growth0[axis];
530  compatibleGrowths[IndexX][IndexY][1][axis]= growth1[axis];
531  compatibleGrowths[IndexX][IndexY][2][axis]= growth2[axis];
532  compatibleGrowths[IndexX][IndexY][3][axis]= growth3[axis];
533  }
534  for (size_t i=0; i<4; ++i){
535  compatibleAngles[IndexX][IndexY][i] = angles[i];
536  compatibleAngleEliminated[IndexX][IndexY][i] = AngleEliminated[i];
537  }
538  } // end of loop over IndexY
539  } //end of loop over IndexX
540  }
